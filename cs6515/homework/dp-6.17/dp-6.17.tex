\section{Exercise 6.17}

\subsection{Problem}

Given an unlimited supply of coins of denominations \(x_1, x_2, \dots, x_n\), we wish to make change for a value \(v\); that is, we wish to find a set of coins whose total value is \(v\). This might not be possible: for instance, if the denominations are 5 and 10 then we can make change for 15 but not for 12. Give an \(O(nv)\) dynamic-programming algorithm for the following problem.
\begin{itemize}[label={}]
    \item \(Input\): \(x_1,\dots,x_n;v\).
    \item \(Question\): Is it possible to make change for \(v\) using coins of denominations \(x_1,\dots,x_n\)?
\end{itemize}

\subsection{Solution}

\begin{algorithm}
    \SetKwInput{KwSubproblem}{Subproblem}
    \SetKwInput{KwRecurrence}{Recurrence}
    \SetKwInput{KwTime}{Time}
    \caption{CC\((C, v)\)}
    \KwIn{A set of \(n\) coin denominations, \(C\), and a currency value, \(v\).}
    \KwOut{A boolean indication if it is possible to make change with a value of \(v\) using an unlimited amount of coins in the set of denominations of \(C\).}
    \KwSubproblem{Let \(M(i)\) = \(true\) if it is possible to make exact change for the value \(i\) using only denominations in \(C\).}
    \KwRecurrence{\(M(i) =
        \begin{cases}
            true,&\text{if }M(i - C_j) = true \text{ for } 1 \le j \le n\\
            false,&\text{otherwise }
        \end{cases}
    \)} 
    \KwTime{O(nV)} 
    \(\space\)\\
    \(M(0) = true\)\\
    \(M(1 \to v) = false\)\\
    \(\space\)\\
    \For {\(i = 1 \to v\)}
    {
        \For {\(j = 1 \to n\)}
        {
            \If{\(C_j <= i \text{ and } M(i-C_j) == true\)}
            {
                \(M(i) = true\)
            }
        }
    }
    \Return\(M(v)\)
\end{algorithm}
